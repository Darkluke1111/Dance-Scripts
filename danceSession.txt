DO
    IF(!@isrunning)
        IF(@&dancing = "stop")
            SPLIT(",",%@&danceoptions%,&dances[])
            ARRAYSIZE(&dances[],#size)
            DEC(#size,1)
            FOR(#i = 0,1,%#size#)
                STOP(%&dances[%#i%]%)
            NEXT
        ELSE        
            LOG("Dancing >%@&dancing%<")
            SET(@isrunning)
            &filename = "%@&dancing%.txt"

            EXEC(%&filename%,%@&dancing%)
        ENDIF
    ENDIF
LOOP
